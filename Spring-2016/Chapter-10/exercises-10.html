fprintf('\n(1;2;0), (1;0;0), (1;0;1).')
u1 = [1;2;0];
u2 = [1;0;0];
u3 = [1;0;1];
fprintf('\nUse routine gschmidt in MATLAB to obtain an orthonormal basis T and')
fprintf('\nthen find the coordinates of')
x = [1;2;3]
fprintf('relative to T. Record the orthonormal basis and the coordinates of')
fprintf('\nx below.')
% MATLAB does not know the function gschmidt. Below is a function found
% online from Yingwei Wang of Purdue University.
function [Q,R] = gschmidt(V)
[m,n] = size(V);
R = zeroes(n);
R(1,1) = norm(V(:,1));
Q(:,1) = V(:,1) / R(1,1);
for k = 2:n
    R(1:k-1, k) = Q(:, 1:k-2)'*V(:,k);
    Q(:,k) = V(:,k) - Q(:,1:k-1) * R(1:k-1,k);
    R(k,k) = norm(Q(:,k));
    Q(:,k) = Q(:,k) / R(k,k);
end
end
A = [u1,u2,u3];
T = gschmidt(A)
